# -*- coding: utf-8 -*-
import spacy

# Загружаем русскую NLP-модель
nlp_ru = spacy.load("ru_core_news_sm")

def process_text(query):
    doc = nlp_ru(query)

    # Фильтруем токены: убираем стоп-слова и оставляем леммы
    keywords = [token.lemma_ for token in doc if not token.is_stop]

    return keywords


query = "уроки по asp.net"
print(process_text(query))
# Выведет: ['урок', 'asp.net']
